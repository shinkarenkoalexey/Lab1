Private Sub CommandButton1_Click()
' Красный
1:
i = Int((40 - 1 + 1) * Rnd + 1)
Set ctrlLabel = Me.Controls("Label" & i)
If ctrlLabel.BackColor = &H80000003 Then
ctrlLabel.BackColor = &HFF&
Else
GoTo 1
End If
End Sub

Private Sub CommandButton10_Click()
' Очистка поля

Next i
End Sub
